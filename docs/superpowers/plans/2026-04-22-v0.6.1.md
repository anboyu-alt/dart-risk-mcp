기존 함수 전체(1308~1372 라인)를 다음으로 교체:

```python
def detect_capital_churn(events: list[dict], lookback_years: int) -> dict:
    """
    12개월 슬라이딩 윈도우 판정:
      (A) 희석성 자본 이벤트 ≥ 3건 → CAPITAL_CHURN
      (B) 희석성 ≥ 2건 AND 비희석성 ≥ 2건 → CAPITAL_CHURN
      그 외(비희석성만 반복, 부족) → 플래그 없음

    Args:
        events: match_signals 반환 이벤트 리스트. 각 항목은
                {"key", "rcept_dt" (YYYYMMDD), "is_amendment", ...}.
        lookback_years: 조회 기간(년). 메타에만 기록.

    Returns:
        {
            "max_12m_count": int,          # 전체 자본 이벤트 12개월 최대 카운트
            "max_dilutive_12m": int,       # 희석성 12개월 최대 카운트
            "max_non_dilutive_12m": int,   # 비희석성 12개월 최대 카운트
            "total_events": int,
            "by_year": dict[str, int],
            "events": list[dict],
            "flags": list[str],
            "lookback_years": int,
        }
    """
    # 1) 자본 이벤트만 필터 + 정정공시 제외
    caps: list[dict] = []
    for e in events or []:
        if e.get("key") not in CAPITAL_EVENT_KEYS:
            continue
        if e.get("is_amendment"):
            continue
        caps.append(e)

    # 2) rcept_dt 오름차순 정렬
    caps.sort(key=lambda e: (e.get("rcept_dt") or "00000000"))

    # 3) 날짜 파싱 (잘못된 포맷은 스킵). 각 이벤트의 희석/비희석 속성 보존.
    parsed: list[tuple[datetime, str, bool]] = []  # (date, key, is_dilutive)
    for e in caps:
        raw = (e.get("rcept_dt") or "")[:8]
        try:
            d = datetime.strptime(raw, "%Y%m%d")
        except ValueError:
            continue
        key = e.get("key") or ""
        parsed.append((d, key, key in DILUTIVE_CAPITAL_EVENTS))

    # 4) 365일 슬라이딩 윈도우에서 전체/희석/비희석 최대 카운트
    max_total = 0
    max_dil = 0
    max_non = 0
    window_hits: list[tuple[int, int]] = []  # (dil, non) per window
    for i, (start, _, _) in enumerate(parsed):
        end = start + timedelta(days=365)
        dil = 0
        non = 0
        for d, _, is_dil in parsed[i:]:
            if start <= d <= end:
                if is_dil:
                    dil += 1
                else:
                    non += 1
        tot = dil + non
        if tot > max_total:
            max_total = tot
        if dil > max_dil:
            max_dil = dil
        if non > max_non:
            max_non = non
        window_hits.append((dil, non))

    # 5) 연도별 집계
    by_year: dict[str, int] = {}
    for e in caps:
        y = (e.get("rcept_dt") or "")[:4]
        if y:
            by_year[y] = by_year.get(y, 0) + 1

    # 6) 플래그 판정 — 단일 윈도우 내에서 조건 (A) 또는 (B) 충족 여부
    flagged = False
    for dil, non in window_hits:
        if dil >= 3:
            flagged = True
            break
        if dil >= 2 and non >= 2:
            flagged = True
            break
    flags = ["CAPITAL_CHURN"] if flagged else []

    return {
        "max_12m_count": max_total,
        "max_dilutive_12m": max_dil,
        "max_non_dilutive_12m": max_non,
        "total_events": len(caps),
        "by_year": by_year,
        "events": caps,
        "flags": flags,
        "lookback_years": lookback_years,
    }
```

```python
"""V.2 FSS 5개 기업의 재무 delta 분포 측정.

임계값 재조정을 위한 1회성 스크립트. 실행 결과를 보고 임계값 결정.
"""
import os
import sys
sys.path.insert(0, os.path.abspath(os.path.join(os.path.dirname(__file__), "..")))

from dart_risk_mcp.core.dart_client import (
    resolve_corp, fetch_financial_statements, _fs_response_to_periods,
    detect_financial_anomaly,
)

API = os.environ.get("DART_API_KEY", "")
if not API:
    print("DART_API_KEY 필요"); sys.exit(1)

# V.2 FSS 사례 5개 — v0.6.0 릴리스 게이트 문서 기준
COMPANIES = [
    "파멥신",      # 자본잠식 사례
    "테라사이언스",  # 반복 CB 발행
    "보타바이오",   # 신사업 허위
    "마이더스AI",   # 감자+CB
    "대유에이피",   # 자산양도
]

rows = []
for name in COMPANIES:
    corp_name, meta = resolve_corp(name, API)
    if not corp_name:
        print(f"[SKIP] {name} corp_code 없음")
        continue
    code = meta.get("corp_code")
    fs = fetch_financial_statements(code, API, "2024", "annual")
    if not fs:
        fs = fetch_financial_statements(code, API, "2023", "annual")
    if not fs:
        print(f"[SKIP] {name} FS 없음")
        continue
    cur, pri = _fs_response_to_periods({"list": fs})
    flags, metrics = detect_financial_anomaly(cur, pri)
    summary = {"name": name, "flags": flags}
    for m in metrics:
        summary[m["name"]] = {
            k: v for k, v in m.items() if k in ("current", "prior", "delta", "unit", "current_ni", "current_ocf")
        }
    rows.append(summary)

import json
print(json.dumps(rows, ensure_ascii=False, indent=2, default=str))
```
