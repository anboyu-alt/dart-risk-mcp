"""유상·유무상증자 공시에서 인수인 목록 추출.

구조화 엔드포인트(`piicDecsn`, `pifricDecsn`)를 사용해
`actnmn`(인수인명), `actsen`(인수인구분), `fric_tisstk_fta`(납입금액) 필드를 읽는다.
무상증자(`fricDecsn`)는 인수인 개념이 없어 여기서 호출하지 않는다.
"""

import re
from .dart_client import fetch_piic_decision, fetch_pifric_decision

__all__ = ["extract_rights_offering_investors"]

_BLANK_PATTERNS = {"", "-", "–", "—", "·", "해당없음", "해당 없음"}


def _clean_name(raw: str) -> str:
    """인수인명 정규화 — 다중 공백 단일화, 양끝 공백 제거."""
    if not raw:
        return ""
    s = re.sub(r"\s+", " ", str(raw)).strip()
    return s


def _parse_list(data: dict) -> list[dict]:
    """DART 응답 dict에서 인수인 레코드 리스트 추출."""
    results = []
    if not data:
        return results
    for row in data.get("list", []) or []:
        name = _clean_name(row.get("actnmn", ""))
        if name in _BLANK_PATTERNS:
            continue
        results.append({
            "name": name,
            "type": (row.get("actsen") or "").strip(),
            "amount": (row.get("fric_tisstk_fta") or row.get("bd_fta") or "").strip(),
            "source": "rights_offering",
        })
    return results


def extract_rights_offering_investors(rcept_no: str, api_key: str) -> list[dict]:
    """
    유상증자·유무상증자 공시에서 인수인 목록 추출.

    Args:
        rcept_no: 공시 접수번호 (14자리)
        api_key: DART API 인증키

    Returns:
        [{"name": 인수인명, "type": 인수인구분, "amount": 납입금액, "source": "rights_offering"}, ...]
        결과가 없으면 빈 리스트.
    """
    # 1) 유상증자 결정 시도
    data = fetch_piic_decision(rcept_no, api_key)
    investors = _parse_list(data)
    if investors:
        return investors

    # 2) 유무상증자 결정 시도
    data = fetch_pifric_decision(rcept_no, api_key)
    investors = _parse_list(data)
    return investors
```

import re

import zipfile
import requests

from .dart_client import (
    fetch_cb_issue_decision,

__all__ = ["extract_cb_investors"]

_DOCUMENT_URL = "https://opendart.fss.or.kr/api/document.xml"
_BLANK_PATTERNS = {"", "-", "–", "—", "·", "해당없음", "해당 없음"}


def _clean_name(raw: str) -> str:
    if not raw:
        return ""
    s = re.sub(r"\s+", " ", str(raw)).strip()
    return s


def _parse_structured(data: dict) -> list[dict]:
    """구조화 응답에서 인수인 필드 추출. 반환 형식은 기존 extract_cb_investors 호환."""
    results = []
    if not data:
        return results
    for row in data.get("list", []) or []:
        name = _clean_name(row.get("actnmn", ""))
        if name in _BLANK_PATTERNS:
            continue
        results.append({
            "name": name,
            "type": (row.get("actsen") or "").strip(),
            "amount": (row.get("bd_fta") or "").strip(),
        })
    return results


def _fetch_text(rcept_no: str, api_key: str) -> str:
    """기존 HTML ZIP 다운로드 로직 — 폴백에서만 사용."""
    params = {"crtfc_key": api_key, "rcept_no": rcept_no}
    for attempt in range(3):
        try:
            resp = requests.get(_DOCUMENT_URL, params=params, timeout=15)
            if resp.status_code == 200:
                break
        except Exception:
            pass
        time.sleep(1 + attempt)
    else:
        return ""

    try:
        zf = zipfile.ZipFile(io.BytesIO(resp.content))
    except zipfile.BadZipFile:
        return ""

    texts = []
    for name in zf.namelist():
        raw = zf.read(name)
        for enc in ("utf-8", "euc-kr", "cp949"):
            try:
                texts.append(raw.decode(enc))
                break
            except UnicodeDecodeError:
                continue
    return "\n".join(texts)


def _legacy_html_extract(rcept_no: str, api_key: str) -> list[dict]:
    """기존 HTML 정규식 기반 추출 — 구조화 엔드포인트 모두 실패 시 폴백."""
    text = _fetch_text(rcept_no, api_key)
    if not text:
        return []
    # 공시 본문에서 '사채인수인' 또는 '인수인' 섹션 주변 테이블 행 탐색
    # 이 정규식은 v0.6.1 이전 동작을 보존한다.
    results = []
    # 인수인명 : XXXX 패턴
    for m in re.finditer(r"인수인\s*[:：]\s*([^\s<>\n]{2,40})", text):
        name = _clean_name(m.group(1))
        if name and name not in _BLANK_PATTERNS:
            results.append({"name": name, "type": "", "amount": ""})
    # 중복 제거 (이름 기준)
    seen = set()
    deduped = []
    for r in results:
        if r["name"] not in seen:
            seen.add(r["name"])
            deduped.append(r)
    return deduped

    """
    CB/BW/EB 발행결정 공시에서 인수인 목록 추출.

    v0.7.0: 구조화 엔드포인트(cvbdIsDecsn / bdwtIsDecsn / exbdIsDecsn) 순차 시도.
    전부 빈 응답·예외일 때 HTML 파싱으로 폴백.

    Returns:
        [{"name": 인수인명, "type": 인수인구분, "amount": 인수금액}, ...]
    """
    for fetch_fn in (
        fetch_cb_issue_decision,
        fetch_bw_issue_decision,
        fetch_eb_issue_decision,
    ):
        try:
            data = fetch_fn(rcept_no, api_key)
        except Exception:
            continue
        investors = _parse_structured(data)
        if investors:
            return investors

    # 구조화 전부 실패 또는 빈 응답 → HTML 폴백
    try:
        return _legacy_html_extract(rcept_no, api_key)
    except Exception:
        return []

def extract_ratios(corp_code, api_key, year):
    """(AR delta %, INV delta %, cash_gap ratio %, capital_impair %) 반환."""
    cur = fetch_financial_statements_all(corp_code, api_key, year=year,
                                          report_type="annual", fs_div="CFS")
    prev = fetch_financial_statements_all(corp_code, api_key, year=str(int(year) - 1),
                                           report_type="annual", fs_div="CFS")
    if not cur or not prev:
        return None

    def find_account(data, keywords):
        for row in data.get("list", []):
            nm = row.get("account_nm", "")
            if any(k in nm for k in keywords):
                val = row.get("thstrm_amount", "").replace(",", "")
                try:
                    return float(val)
                except ValueError:
                    return None
        return None

    ar_cur = find_account(cur, ["매출채권"])
    ar_prev = find_account(prev, ["매출채권"])
    inv_cur = find_account(cur, ["재고자산"])
    inv_prev = find_account(prev, ["재고자산"])
    op_cash = find_account(cur, ["영업활동", "현금흐름"])
    net_inc = find_account(cur, ["당기순이익"])
    equity = find_account(cur, ["자본총계"])
    capital = find_account(cur, ["자본금"])

    def pct_delta(c, p):
        if c is None or p is None or p == 0:
            return None
        return (c - p) / abs(p) * 100

    return {
        "ar_delta_pct": pct_delta(ar_cur, ar_prev),
        "inv_delta_pct": pct_delta(inv_cur, inv_prev),
        "cash_gap_ratio_pct": (op_cash / net_inc * 100)
            if (op_cash is not None and net_inc not in (None, 0)) else None,
        "capital_impair_pct": (equity / capital * 100)
            if (equity is not None and capital not in (None, 0)) else None,
    }


def main():
    api_key = os.environ.get("DART_API_KEY")
    if not api_key:
        print("ERROR: DART_API_KEY not set"); sys.exit(1)

    # 수집
    candidates = collect_candidates(api_key, days=365)
    print(f"후보 기업 수: {len(candidates)}")

    year = time.strftime("%Y", time.gmtime(time.time() - 90 * 86400))  # 직전 연도

    rows = []
    for corp_code, corp_name in candidates[:40]:  # 상한
        r = extract_ratios(corp_code, api_key, year)
        if not r:
            continue
        rows.append((corp_name, r))
        if len(rows) >= 20:
            break
        time.sleep(0.3)

    print(f"유효 샘플: {len(rows)}")

    # 분포 출력
    import statistics
    for metric in ("ar_delta_pct", "inv_delta_pct",
                    "cash_gap_ratio_pct", "capital_impair_pct"):
        vals = [r[metric] for _, r in rows if r[metric] is not None]
        if not vals:
            print(f"{metric}: no data"); continue
        print(f"\n{metric}: n={len(vals)}")
        print(f"  min={min(vals):.1f}  median={statistics.median(vals):.1f}  "
              f"max={max(vals):.1f}")
        q = statistics.quantiles(vals, n=4) if len(vals) >= 4 else vals
        print(f"  quartiles: {[round(x, 1) for x in q]}")

    # 샘플 회사별 raw
    print("\n=== raw ===")
    for name, r in rows:
        print(f"{name}: AR={r['ar_delta_pct']} INV={r['inv_delta_pct']} "
              f"CASH={r['cash_gap_ratio_pct']} IMPAIR={r['capital_impair_pct']}")


if __name__ == "__main__":
    main()
```
